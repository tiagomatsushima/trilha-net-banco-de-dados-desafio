    F.Nome AS Nome, 
    A.PrimeiroNome, 
    A.UltimoNome, 
    EF.Papel
FROM 
    Filmes F

    FilmesGenero FG ON F.Id = FG.IdFilme

    Generos G ON FG.IdGenero = G.Id
JOIN 
    ElencoFilme EF ON F.Id = EF.IdFilme

    Atores A ON EF.IdAtor = A.Id;
